    lui x1, 0xFFFF0      # x1 = 0xFFFF0000 (LED 位址)
    li  x2, 0x55         # pattern 1
    li  x3, 0xAA         # pattern 2

loop:
    # --- LED = 0x55 ---
    sw   x2, 0(x1)       # 寫入 LED

    # delay 約 1 秒
    li   x4, 25000000    # N = 25,000,000
delay1:
    addi x4, x4, -1
    bnez x4, delay1

    # --- LED = 0xAA ---
    sw   x3, 0(x1)       # 寫入 LED

    # delay 約 1 秒
    li   x4, 25000000
delay2:
    addi x4, x4, -1
    bnez x4, delay2

    j    loop
